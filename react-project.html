React Task Manager

A full-featured task management application built with React 18 — leveraging hooks, Context API, and localStorage persistence for a smooth, stateful user experience.

React 18 JavaScript Context API Hooks localStorage Vite

Project Overview

React Task Manager demonstrates idiomatic React 18 patterns — functional components throughout, custom hooks for business logic, and the Context + useReducer combo as a lightweight Redux alternative.

Tasks are persisted in localStorage via a dedicated usePersist hook, so data survives page refreshes without any backend.

The project is scaffolded with Vite for instant hot-module replacement and an optimised production build, keeping development fast and the final bundle lean.

Code Examples

Task Context + useReducer

React / JSX 
import { createContext, useContext, useReducer } from 'react';

const TaskContext = createContext(null);

function taskReducer(state, action) {
    switch (action.type) {
        case 'ADD':
            return [...state, { id: Date.now(), text: action.payload, done: false }];
        case 'TOGGLE':
            return state.map(t => t.id === action.id ? { ...t, done: !t.done } : t);
        case 'DELETE':
            return state.filter(t => t.id !== action.id);
        case 'EDIT':
            return state.map(t => t.id === action.id ? { ...t, text: action.text } : t);
        default: return state;
    }
}

export function TaskProvider({ children }) {
    const [tasks, dispatch] = useReducer(taskReducer, [], init);
    return (
        <TaskContext.Provider value={{ tasks, dispatch }}>
            {children}
        </TaskContext.Provider>
    );
}

export const useTasks = () => useContext(TaskContext);

usePersist — localStorage Sync Hook

React / JSX 
import { useEffect } from 'react';

/**
 * Persists `value` to localStorage under `key` on every change.
 * Pass as the `init` function to useReducer to hydrate on mount.
 */
export function usePersist(key, value) {
    useEffect(() => {
        try {
            localStorage.setItem(key, JSON.stringify(value));
        } catch (e) {
            // quota exceeded or private mode — fail silently
        }
    }, [key, value]);
}

// Initialiser passed to useReducer — reads from storage once on mount
function init() {
    try {
        const stored = localStorage.getItem('tasks');
        return stored ? JSON.parse(stored) : [];
    } catch { return []; }
}

TaskItem — Toggle, Edit, Delete

React / JSX 
export function TaskItem({ task }) {
    const { dispatch } = useTasks();
    const [editing, setEditing] = useState(false);
    const [draft, setDraft]     = useState(task.text);

    const commitEdit = () => {
        if (draft.trim()) dispatch({ type: 'EDIT', id: task.id, text: draft.trim() });
        setEditing(false);
    };

    return (
        <li className={`task-item ${task.done ? 'done' : ''}`}>
            <input type="checkbox" checked={task.done}
                   onChange={() => dispatch({ type: 'TOGGLE', id: task.id })} />
            {editing
                ? <input value={draft} onChange={e => setDraft(e.target.value)}
                          onBlur={commitEdit} onKeyDown={e => e.key === 'Enter' && commitEdit()} autoFocus />
                : <span onDoubleClick={() => setEditing(true)}>{task.text}</span>
            }
            <button onClick={() => dispatch({ type: 'DELETE', id: task.id })}>✕</button>
        </li>
    );
}
